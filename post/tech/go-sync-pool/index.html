what is sync.Pool?#

从名称中可以看出，这是一个池，和常见的池概念一样，它可以通过重复利用对象，从而减少内存分配、减少gc耗时来达到提高程序性能的目的。

source code of sync.Pool#

type Pool struct {
	// nocopy机制，用于go vet命令检查是否复制后使用
	noCopy noCopy
	// local fixed-size per-P pool, actual type is [P]poolLocal
	// 指向[P]poolLocal数组，该P即GMP中P的数量，每一个P都有对应的poolLocal，P 的 id 对应 [P]poolLocal 下标索引
	local     unsafe.Pointer
	localSize uintptr        // size of the local array

	victim     unsafe.Pointer // local from previous cycle
	victimSize uintptr        // size of victims array

	// New optionally specifies a function to generate
	// a value when Get would otherwise return nil.
	// It may not be changed concurrently with calls to Get.
	// 用于在Get返回nil时创建对象，不能在Get调用的同时修改
	New func() any
}

type poolLocal struct {
	poolLocalInternal

	// Prevents false sharing on widespread platforms with
	// 128 mod (cache line size) = 0 .
	// 避免伪共享
	pad [128 - unsafe.Sizeof(poolLocalInternal{})%128]byte
}

type poolLocalInternal struct {
	private any       // Can be used only by the respective P. P 的私有缓存区，使用时无需要加锁
	shared  poolChain // Local P can pushHead/popHead; any P can popTail. 公共缓存区。本地 P 可以 pushHead/popHead；其他 P 则只能 popTail
}

type poolChain struct {
	// head is the poolDequeue to push to. This is only accessed
	// by the producer, so doesn't need to be synchronized.
	head *poolChainElt

	// tail is the poolDequeue to popTail from. This is accessed
	// by consumers, so reads and writes must be atomic.
	tail *poolChainElt
}

poolLocalInternal中的shared字段指向一个双向链表poolChain，链表的节点类型为poolChainElt，其中包含的poolDequeue是一个双端队列(Double-ended Queue)

type poolChainElt struct {
	poolDequeue

	// next and prev link to the adjacent poolChainElts in this
	// poolChain.
	//
	// next is written atomically by the producer and read
	// atomically by the consumer. It only transitions from nil to
	// non-nil.
	//
	// prev is written atomically by the consumer and read
	// atomically by the producer. It only transitions from
	// non-nil to nil.
	next, prev *poolChainElt
}

type poolDequeue struct {

	headTail uint64

	// vals is a ring buffer of interface{} values stored in this
	// dequeue. ***The size of this must be a power of 2***.
	vals []eface
}

operation of sync.Pool#

Get#

// Get selects an arbitrary item from the Pool, removes it from the
// Pool, and returns it to the caller.
// Get may choose to ignore the pool and treat it as empty.
// Callers should not assume any relation between values passed to Put and
// the values returned by Get.
//
// If Get would otherwise return nil and p.New is non-nil, Get returns
// the result of calling p.New.
func (p *Pool) Get() any {
	// 竞态分析相关
	if race.Enabled {
		race.Disable()
	}
	// pin将当前g与P绑定，不允许抢占，并返回该P的poolLocal和id
	l, pid := p.pin()
	x := l.private
	l.private = nil
	if x == nil {
		// 尝试从本地头部pop一个对象，如果仍为nil，使用getSlow方法尝试从其他P的shared双端队列尾部“偷”一个对象出来
		x, _ = l.shared.popHead()
		if x == nil {
			x = p.getSlow(pid)
		}
	}
	// 解除pin的绑定
	runtime_procUnpin()
	// 竞态分析相关
	if race.Enabled {
		race.Enable()
		if x != nil {
			race.Acquire(poolRaceAddr(x))
		}
	}
	// 如果还是没取到缓存对象，使用New创建一个
	if x == nil && p.New != nil {
		x = p.New()
	}
	return x
}

Get会执行pin方法，绑定当前*goroutine*和P，防止抢占，导致P的更换，并返回该P的poolLocal和pid，如果该poolLocal的private不为空，可置空后直接返回，如果为空，则按照以下步骤尝试获取：

  1. 从本地队列pop一个对象
  2. 从其他P的队列“偷”一个对象
  3. 尝试从victim cache中获取
  4. 以上都没法获取到，如果有New方法，调用New方法创建

看看其中的关键函数

pin:

// pin pins the current goroutine to P, disables preemption and
// returns poolLocal pool for the P and the P's id.
// Caller must call runtime_procUnpin() when done with the pool.
func (p *Pool) pin() (*poolLocal, int) {
	pid := runtime_procPin()
	// In pinSlow we store to local and then to localSize, here we load in opposite order.
	// Since we've disabled preemption, GC cannot happen in between.
	// Thus here we must observe local at least as large localSize.
	// We can observe a newer/larger local, it is fine (we must observe its zero-initialized-ness).

	// runtime/internal/atomic.LoadAcquintptr
	s := runtime_LoadAcquintptr(&p.localSize) // load-acquire 
	l := p.local                              // load-consume
	// 如果localSize大于pid，则直接通过索引偏移获取poolLocal
	if uintptr(pid) < s {
		return indexLocal(l, pid), pid
	}
	// 当数组大小不够时，调用pinSlow
	return p.pinSlow()
}

//go:nosplit
//go:noinline
func LoadAcquintptr(ptr *uintptr) uintptr {
	return *ptr
}

func indexLocal(l unsafe.Pointer, i int) *poolLocal {
	lp := unsafe.Pointer(uintptr(l) + uintptr(i)*unsafe.Sizeof(poolLocal{}))
	return (*poolLocal)(lp)
}

pinSlow:

func (p *Pool) pinSlow() (*poolLocal, int) {
	// Retry under the mutex.
	// Can not lock the mutex while pinned.
	runtime_procUnpin()

	allPoolsMu.Lock()
	defer allPoolsMu.Unlock()
	// 再次pin
	pid := runtime_procPin()
	// poolCleanup won't be called while we are pinned.
	// 重新获取，因为执行过程中它们可能被修改了，这边已经上锁，所以不需要进行原子性的读取操作
	s := p.localSize
	l := p.local
	// 重新判断，因为此时的P可能不再是之前绑定那个P
	if uintptr(pid) < s {
		return indexLocal(l, pid), pid
	}
	// local为空，该pool的第一次Get
	if p.local == nil {
		allPools = append(allPools, p)
	}
	// If GOMAXPROCS changes between GCs, we re-allocate the array and lose the old one.
	size := runtime.GOMAXPROCS(0)
	// 重新创建一个poolLocal数组
	local := make([]poolLocal, size)
	atomic.StorePointer(&p.local, unsafe.Pointer(&local[0])) // store-release
	runtime_StoreReluintptr(&p.localSize, uintptr(size))     // store-release
	return &local[pid], pid
}

可以看到在pinSlow的第一行进行了pin的解除操作，这是因为接下来的操作会需要获取allPoolsMu锁，而获取该锁的耗时可能会很长，为了避免长时间占用P，所以先解除当前pin操作。

为什么，在Pool中的poolLocal数量不满足P的数量时，要重新创建poolLocal数组？

poolChain.popHead:

func (c *poolChain) popHead() (any, bool) {
	// 取出链表头
	d := c.head
	for d != nil {
		if val, ok := d.popHead(); ok {
			return val, ok
		}
		// There may still be unconsumed elements in the
		// previous dequeue, so try backing up.
		// 取链表的上一个节点，继续尝试
		d = loadPoolChainElt(&d.prev)
	}
	return nil, false
}

poolDequeue.popHead:

func (d *poolDequeue) popHead() (any, bool) {
	var slot *eface
	for {
		// 原子读取headTail
		ptrs := atomic.LoadUint64(&d.headTail)
		// 解包-见右部
		head, tail := d.unpack(ptrs)

		if tail == head {
			// 队列为空.
			return nil, false
		}

		// Confirm tail and decrement head. We do this before
		// reading the value to take back ownership of this
		// slot.
		head--
		ptrs2 := d.pack(head, tail)
		// CAS操作
		if atomic.CompareAndSwapUint64(&d.headTail, ptrs, ptrs2) {
			// We successfully took back slot.
			slot = &d.vals[head&uint32(len(d.vals)-1)]
			break
		}
	}

	val := *(*any)(unsafe.Pointer(slot))
	if val == dequeueNil(nil) {
		val = nil
	}
	// Zero the slot. Unlike popTail, this isn't racing with
	// pushHead, so we don't need to be careful here.
	*slot = eface{}
	return val, true
}

const dequeueBits = 32

func (d *poolDequeue) unpack(ptrs uint64) (head, tail uint32) {
	// mask的右32位都为1
	const mask = 1<<dequeueBits - 1
	// 通过位运算取得head、tail
	// 即ptrs的前32位为head，后32位为tail
	head = uint32((ptrs >> dequeueBits) & mask)
	tail = uint32(ptrs & mask)
	return
}

func (d *poolDequeue) pack(head, tail uint32) uint64 {
	const mask = 1<<dequeueBits - 1
	return (uint64(head) << dequeueBits) |
		uint64(tail&mask)
}

// dequeueNil is used in poolDequeue to represent interface{}(nil).
// Since we use nil to represent empty slots, we need a sentinel value
// to represent nil.
type dequeueNil *struct{}

getSlow:


func (p *Pool) getSlow(pid int) any {
	// See the comment in pin regarding ordering of the loads.
	// 原子操作获取localSize
	size := runtime_LoadAcquintptr(&p.localSize) // load-acquire
	// locals——poolLocal数组
	locals := p.local                            // load-consume
	// Try to steal one element from other procs.
	for i := 0; i < int(size); i++ {
		l := indexLocal(locals, (pid+i+1)%int(size))
		if x, _ := l.shared.**popTail**(); x != nil {
			return x
		}
	}

	// Try the victim cache. We do this after attempting to steal
	// from all primary caches because we want objects in the
	// victim cache to age out if at all possible.
	// 尝试从victim cache中获取，如果未获取到，将其Size置0，下次不用再找
	size = atomic.LoadUintptr(&p.victimSize)
	if uintptr(pid) >= size {
		return nil
	}
	locals = p.victim
	l := indexLocal(locals, pid)
	if x := l.private; x != nil {
		l.private = nil
		return x
	}
	for i := 0; i < int(size); i++ {
		l := indexLocal(locals, (pid+i)%int(size))
		if x, _ := l.shared.popTail(); x != nil {
			return x
		}
	}

	// Mark the victim cache as empty for future gets don't bother
	// with it.
	atomic.StoreUintptr(&p.victimSize, 0)

	return ni

getSlow方法是从其他的P中窃取对象，获取poolLocal数组的大小后，进行遍历，调用其他poolLocal的popTail方法，如果成功获取到则立即返回。

如果未找到，尝试从victim cache中获取，victim cache的找寻逻辑和poolLocal一样。

接下来看看popTail方法:

func (c *poolChain) popTail() (any, bool) {
	d := loadPoolChainElt(&c.tail)
	if d == nil {
		return nil, false
	}

	for {
		// It's important that we load the next pointer
		// *before* popping the tail. In general, d may be
		// transiently empty, but if next is non-nil before
		// the pop and the pop fails, then d is permanently
		// empty, which is the only condition under which it's
		// safe to drop d from the chain.
		// 在pop之前加载next指针很重要，通常来说，d可能暂时为空，
		// 但是，当d2不为空且d的pop失败，那d将会永远为空，而这是唯一可以
		// 从链表中删除d的条件
		d2 := loadPoolChainElt(&d.next)

		if val, ok := d.popTail(); ok {
			return val, ok
		}

		if d2 == nil {
			// This is the only dequeue. It's empty right
			// now, but could be pushed to in the future.
			return nil, false
		}

		// The tail of the chain has been drained, so move on
		// to the next dequeue. Try to drop it from the chain
		// so the next pop doesn't have to look at the empty
		// dequeue again.
		// 移动tail指针到d2
		if atomic.CompareAndSwapPointer((*unsafe.Pointer)(unsafe.Pointer(&c.tail)), unsafe.Pointer(d), unsafe.Pointer(d2)) {
			// We won the race. Clear the prev pointer so
			// the garbage collector can collect the empty
			// dequeue and so popHead doesn't back up
			// further than necessary.
			storePoolChainElt(&d2.prev, nil)
		}
		d = d2
	}
}

Put#

// Put adds x to the pool.
func (p *Pool) Put(x any) {
	if x == nil {
		return
	}
	if race.Enabled {
		if fastrandn(4) == 0 {
			// Randomly drop x on floor.
			return
		}
		race.ReleaseMerge(poolRaceAddr(x))
		race.Disable()
	}
	l, _ := p.pin()
	if l.private == nil {
		l.private = x
		x = nil
	}
	if x != nil {
		l.shared.pushHead(x)
	}
	runtime_procUnpin()
	if race.Enabled {
		race.Enable()
	}
}

首先，nil会直接return，不会进行任何操作。忽略竞态相关代码，执行pin方法获取localPool，优先存储在private字段，其次是使用pushHead方法存贮。

func (c *poolChain) pushHead(val any) {
	d := c.head
	if d == nil {
		// Initialize the chain.
		const initSize = 8 // Must be a power of 2
		d = new(poolChainElt)
		d.vals = make([]eface, initSize)
		c.head = d
		storePoolChainElt(&c.tail, d)
	}

	if d.pushHead(val) {
		return
	}

	// The current dequeue is full. Allocate a new one of twice
	// the size.
	newSize := len(d.vals) * 2
	if newSize >= dequeueLimit {
		// Can't make it any bigger.
		newSize = dequeueLimit
	}

	d2 := &poolChainElt{prev: d}
	d2.vals = make([]eface, newSize)
	c.head = d2
	storePoolChainElt(&d.next, d2)
	d2.pushHead(val)
}

poolChain先获取head，如果head为空，则该链表需要进行初始化。初始化完成后，调用head的poolChainElt.pushHead方法。

如果poolChainElt.pushHead调用成功，将会直接return，而如果调用失败，说明当前poolChain中的第一个双端队列已经满了，于是创建第二个，容量是上一个的两倍，最大为dequeueLimit。

新的双端队列d2将被设置为head。

// pushHead adds val at the head of the queue. It returns false if the
// queue is full. It must only be called by a single producer.
func (d *poolDequeue) pushHead(val any) bool {
	ptrs := atomic.LoadUint64(&d.headTail)
	head, tail := d.unpack(ptrs)
	if (tail+uint32(len(d.vals)))&(1<<dequeueBits-1) == head {
		// Queue is full.
		return false
	}
	slot := &d.vals[head&uint32(len(d.vals)-1)]

	// Check if the head slot has been released by popTail.
	typ := atomic.LoadPointer(&slot.typ)
	if typ != nil {
		// Another goroutine is still cleaning up the tail, so
		// the queue is actually still full.
		return false
	}

	// The head slot is free, so we own it.
	if val == nil {
		val = dequeueNil(nil)
	}
	*(*any)(unsafe.Pointer(slot)) = val

	// Increment head. This passes ownership of slot to popTail
	// and acts as a store barrier for writing the slot.
	atomic.AddUint64(&d.headTail, 1<<dequeueBits)
	return true
}

poolDequeue.pushHead方法。和popHead中的操作类似，unpack获取head和tail，判断队列是否已满，这里的判断条件是(tail+uint32(len(d.vals)))&(1<<dequeueBits-1) == head，

如果是第一次初始化，len(d.vals)为8，tail为0，当push成功时，head会递增，第一次判断成功会在head等于8时。

head的递增语句是atomic.AddUint64(&d.headTail, 1<<dequeueBits)，加的是1«32，相当于headTail的前32位上+1。

	typ := atomic.LoadPointer(&slot.typ)
	if typ != nil {
		return false
	}

上面这部分代码和popTail有关系，下面详细说明

图中框出的分别是popHead方法和popTail方法在取对象时的代码，popHead的调用者是当前P，它不会和pushHead并发调用，而popTail是由另外的P调用，所以可能会和当前P的pushHead竞争。

popTail在取数完成后，先置solt的val为nil，再原子性修改typ，不同与pushHead，pushHead是没有竞态问题，所以可以直接将其赋值为eface{}。

*slot = eface{}或slot = *eface(nil)不是一个原子操作。因为每个槽位存放2个8字节的unsafe.Pointer。而Go atomic包是不支持16字节原子操作，只能原子性操作solt中的其中一个字段。

如果在popTail的同时pushHead获取了该solt，判断到typ不为nil，说明该solt的控制权还是属于popTail，则认为队列仍满。

Pool回收#

func poolCleanup() {
	// This function is called with the world stopped, at the beginning of a garbage collection.
	// It must not allocate and probably should not call any runtime functions.

	// Because the world is stopped, no pool user can be in a
	// pinned section (in effect, this has all Ps pinned).

	// Drop victim caches from all pools.
	for _, p := range oldPools {
		p.victim = nil
		p.victimSize = 0
	}

	// Move primary cache to victim cache.
	for _, p := range allPools {
		p.victim = p.local
		p.victimSize = p.localSize
		p.local = nil
		p.localSize = 0
	}

	// The pools with non-empty primary caches now have non-empty
	// victim caches and no pools have primary caches.
	oldPools, allPools = allPools, nil
}

// Hooks for other packages

var poolcleanup func()

//go:linkname sync_runtime_registerPoolCleanup sync.runtime_registerPoolCleanup
func sync_runtime_registerPoolCleanup(f func()) {
	poolcleanup = f
}

func clearpools() {
	// clear sync.Pools
	**if poolcleanup != nil {
		poolcleanup()
	}**

	// Clear central sudog cache.
	// Leave per-P caches alone, they have strictly bounded size.
	// Disconnect cached list before dropping it on the floor,
	// so that a dangling ref to one entry does not pin all of them.
	lock(&sched.sudoglock)
	var sg, sgnext *sudog
	for sg = sched.sudogcache; sg != nil; sg = sgnext {
		sgnext = sg.next
		sg.next = nil
	}
	sched.sudogcache = nil
	unlock(&sched.sudoglock)

	// Clear central defer pool.
	// Leave per-P pools alone, they have strictly bounded size.
	lock(&sched.deferlock)
	// disconnect cached list before dropping it on the floor,
	// so that a dangling ref to one entry does not pin all of them.
	var d, dlink *_defer
	for d = sched.deferpool; d != nil; d = dlink {
		dlink = d.link
		d.link = nil
	}
	sched.deferpool = nil
	unlock(&sched.deferlock)
}

func gcStart(trigger gcTrigger) {
	...
	clearpools()
	...
}

poolCleanup这一段代码很容易理解，重点是接下来的部分，也就是什么时候调用poolCleanup

func init() {
	runtime_registerPoolCleanup(poolCleanup)
}

在init函数中使用runtime_registerPoolCleanup函数注册poolCleanup为GC时的回调。runtime_registerPoolCleanup的实现在runtime/mgc.go包中。

在sync包中调用runtime_registerPoolCleanup实际上调用到了左侧的sync_runtime_registerPoolCleanup，接下来的重点在clearpools中，该函数在gcStart函数中被调用。
