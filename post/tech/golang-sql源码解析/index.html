抽象定义#

database/sql库是go标准库的一部分，它本身不提供数据库驱动，而是定义了一套轻量级的接口来与SQL或类SQL的数据库进行交互，由如下核心接口组成：

Connector#

Connector是连接器的抽象，用于获取连接。

// /src/database/sql/driver/driver.go:121
// 一个Connector代表一个固定配置的驱动，能够创建任意数量的、等价的连接给多个协程使用
//
// Connector可以传递给sql.OpenDB函数，让每种数据库实现各自的*sql.DB
//
// 如果Connector实现了io.Closer，会在sql.DB的Close方法最后调用Close()
type Connector interface {
	// Connect 返回一个数据库连接
	// sql库有空闲连接池机制，用于连接重用
	//
	// context仅用于拨号目的（见 net.DialContext），不应存储或用于其他目的。
    // 一个默认的超时时间是必要的，因为连接池可能会异步调用Connect方法来处理查询
	//
	// 返回的连接同一时刻只能给一个goroutine使用
	Connect(context.Context) (Conn, error)

	// Driver方法返回Connector的底层Driver，主要是为了与sql.DB上的Driver方法保持兼容。
	Driver() Driver
}

Stmt 预处理状态#

// /src/database/sql/driver/driver.go:329
// Stmt是一个预处理语句。它绑定到一个Conn，不会被多个goroutine同时使用。
type Stmt interface {
	// Close closes the statement.
	Close() error

	// NumInput 返回占位符参数的数量
	//
	// 如果占位符参数大于0, sql包会对调用者的参数数量进行健全性检查，
	// 如果不匹配，在调用Exec和Query之前报错
    //
	// 如果driver无法计算出占位符数量，NumInput也可以返回-1, 此时sql包不会进行参数数量检查
	NumInput() int

	// Exec 执行Insert或Update
	//
	// Deprecated: Drivers should implement StmtExecContext instead (or additionally).
	Exec(args []Value) (Result, error)

	// Query 执行Select
	//
	// Deprecated: Drivers should implement StmtQueryContext instead (or additionally).
	Query(args []Value) (Rows, error)
}

Rows 查询的结果#

// /src/database/sql/driver/driver.go:423
type Rows interface {
	// Columns返回列的名称。结果的列数就是切片的长度
    // 如果某个特定列名未知，也应有一个空字符串占位
	Columns() []string

	Close() error

	// 每次调用Next，往dest中填充一行数据，dest的切片长度应该和列数保持一致

	// should be taken when closing Rows not to modify

	Next(dest []Value) error
}

实体定义#

除了定义以上的抽象接口，sql库还定义并实现了一些实体类型，主要是用于实现连接池逻辑和对抽象结构的再封装

sql.DB#

最核心的类是sql.DB，对应为数据库的具象化实例，其中包含如下几个核心字段：

// /src/database/sql/sql.go:455
// 
type DB struct {
	// 等待新连接的总耗时
	waitDuration atomic.Int64

    // 上文介绍的抽象连接器
	connector driver.Connector
	// 记录已关闭连接的数量

	connRequests map[uint64]chan connRequest // 用于唤醒阻塞等待连接的goroutine的channel
	nextRequest  uint64 // 维护全局递增计数器，作为connRequests的key
	numOpen      int    // 已开启使用或等待使用的连接数量
	// Used to signal the need for new connections
	// a goroutine running connectionOpener() reads on this chan and
	// maybeOpenNewConnections sends on the chan (one send per needed connection)
	// It is closed during db.Close(). The close tells the connectionOpener
	// goroutine to exit.
	openerCh          chan struct{} // 用于向 opener 传递创建连接信号的 chan
	closed            bool
	dep               map[finalCloser]depSet // 在关闭数据库前，进行依赖梳理
	lastPut           map[*driverConn]string // stacktrace of last conn's put; debug only
	maxIdleCount      int                    // zero means defaultMaxIdleConns; negative means 0
	maxOpen           int                    // <= 0 means unlimited
	maxLifetime       time.Duration          // maximum amount of time a connection may be reused
	maxIdleTime       time.Duration          // maximum amount of time a connection may be idle before being closed
	cleanerCh         chan struct{}
	waitCount         int64
	maxIdleClosed     int64
	maxIdleTimeClosed int64
	maxLifetimeClosed int64

	stop func() // stop cancels the connection opener.
}
  • connector：用于创建数据库连接的抽象连接器，由第三方数据库提供具体实现
  • mu：互斥锁，保证并发安全
  • freeConn：数据库连接池，缓存可用的连接以供后续复用
  • connRequests：唤醒通道集合，和阻塞等待连接的协程是一对一的关系
  • openerCh：创建连接信号通道. 用于向连接创建协程 opener goroutine 发送信号
  • stop：连接创建协程 opener goroutine 的终止器，用于停止该协程

type Tx struct {

	// is an active query. It is held for read during queries

		sync.Mutex

创建数据库对象#

// /src/database/sql/sql.go:813
func Open(driverName, dataSourceName string) (*DB, error) {
    /* 检查是否注册了driverName对应的驱动 */
	driversMu.RLock()
	driveri, ok := drivers[driverName]
	driversMu.RUnlock()
	if !ok {
		return nil, fmt.Errorf("sql: unknown driver %q (forgotten import?)", driverName)
	}
    /* 检查是否注册了driverName对应的驱动 */
	
    // 如果driver实现了driver.DriverContext，则会调用OpenConnector获取Connector，否则默认使用dsnConnector
	if driverCtx, ok := driveri.(driver.DriverContext); ok {
        // 从dataSourceName中解析config，构建一个connector
		connector, err := driverCtx.OpenConnector(dataSourceName)
		if err != nil {
			return nil, err
		}
		return OpenDB(connector), nil
	}
    // 默认是dsnConnector
	return OpenDB(dsnConnector{dsn: dataSourceName, driver: driveri}), nil
}

// 与mysql建立连接，创建mysqlConn
func (c *connector) Connect(ctx context.Context) (driver.Conn, error) {
	var err error

	// New mysqlConn
	mc := &mysqlConn{
		maxAllowedPacket: maxPacketSize,
		maxWriteSize:     maxPacketSize - 1,
		closech:          make(chan struct{}),
		cfg:              c.cfg,
	}
	mc.parseTime = mc.cfg.ParseTime

	// Connect to Server
	dialsLock.RLock()
	dial, ok := dials[mc.cfg.Net]
	dialsLock.RUnlock()
	if ok {
		dctx := ctx
		if mc.cfg.Timeout > 0 {
			var cancel context.CancelFunc
			dctx, cancel = context.WithTimeout(ctx, c.cfg.Timeout)
			defer cancel()
		}
		mc.netConn, err = dial(dctx, mc.cfg.Addr)
	} else {
		nd := net.Dialer{Timeout: mc.cfg.Timeout}
		mc.netConn, err = nd.DialContext(ctx, mc.cfg.Net, mc.cfg.Addr)
	}

	if err != nil {
		return nil, err
	}

	// Enable TCP Keepalives on TCP connections
	if tc, ok := mc.netConn.(*net.TCPConn); ok {
		if err := tc.SetKeepAlive(true); err != nil {
			// Don't send COM_QUIT before handshake.
			mc.netConn.Close()
			mc.netConn = nil
			return nil, err
		}
	}

	// Call startWatcher for context support (From Go 1.8)
	mc.startWatcher()
	if err := mc.watchCancel(ctx); err != nil {
		mc.cleanup()
		return nil, err
	}
	defer mc.finish()

	mc.buf = newBuffer(mc.netConn)

	// Set I/O timeouts
	mc.buf.timeout = mc.cfg.ReadTimeout
	mc.writeTimeout = mc.cfg.WriteTimeout

	// Reading Handshake Initialization Packet
	authData, plugin, err := mc.readHandshakePacket()
	if err != nil {
		mc.cleanup()
		return nil, err
	}

	if plugin == "" {
		plugin = defaultAuthPlugin
	}

	// Send Client Authentication Packet
	authResp, err := mc.auth(authData, plugin)
	if err != nil {
		// try the default auth plugin, if using the requested plugin failed
		errLog.Print("could not use requested auth plugin '"+plugin+"': ", err.Error())
		plugin = defaultAuthPlugin
		authResp, err = mc.auth(authData, plugin)
		if err != nil {
			mc.cleanup()
			return nil, err
		}
	}
	if err = mc.writeHandshakeResponsePacket(authResp, plugin); err != nil {
		mc.cleanup()
		return nil, err
	}

	// Handle response to auth packet, switch methods if possible
	if err = mc.handleAuthResult(authData, plugin); err != nil {
		// Authentication failed and MySQL has already closed the connection
		// (https://dev.mysql.com/doc/internals/en/authentication-fails.html).
		// Do not send COM_QUIT, just cleanup and return the error.
		mc.cleanup()
		return nil, err
	}

	if mc.cfg.MaxAllowedPacket > 0 {
		mc.maxAllowedPacket = mc.cfg.MaxAllowedPacket
	} else {
		// Get max allowed packet size
		maxap, err := mc.getSystemVar("max_allowed_packet")
		if err != nil {
			mc.Close()
			return nil, err
		}
		mc.maxAllowedPacket = stringToInt(maxap) - 1
	}
	if mc.maxAllowedPacket < maxPacketSize {
		mc.maxWriteSize = mc.maxAllowedPacket
	}

	// Handle DSN Params
	err = mc.handleParams()
	if err != nil {
		mc.Close()
		return nil, err
	}

	return mc, nil
}

OpenDB函数：

// /src/database/sql/sql.go:781
// OpenDB可能仅仅只验证参数，不保证实际创建一个连接，如果要验证dataSourceName是否可用，用Ping
// 返回的数据库可以安全地供多个goroutine并发使用，并维护其自己的空闲连接池。一般只调用一次OpenDB函数。很少需要关闭一个DB再创建
func OpenDB(c driver.Connector) *DB {
	ctx, cancel := context.WithCancel(context.Background())
	db := &DB{
		connector:    c,
		openerCh:     make(chan struct{}, connectionRequestQueueSize), // connectionRequestQueueSize默认100w
		lastPut:      make(map[*driverConn]string),
		connRequests: make(map[uint64]chan connRequest),
		stop:         cancel,
	}
	
   	// 创建了一个协程用于获取连接
	go db.connectionOpener(ctx)

	return db
}

创建连接#

OpenDB会启动一个 connectionOpener 协程，用于在连接池资源不足时，创建连接进行补充，用到了context进行并发控制,其具体行为：

// /src/database/sql/sql.go:1216
func (db *DB) connectionOpener(ctx context.Context) {
	for {
		select {
		case <-ctx.Done():
			return
		case <-db.openerCh:
			db.openNewConnection(ctx)
		}
	}
}

启动了一个for循环，主要监听db.openerCh，当收到信号时调用db.openNewConnection创建新连接，创建连接的代码在openNewConnection中：

// /src/database/sql/sql.go:1228
func (db *DB) openNewConnection(ctx context.Context) {
	// maybeOpenNewConnections 在send db.openerCh之前已经执行了db.numOpen++
    // 所以这里失败了的话，必须执行db.numOpen--
    
    // 调用第三方驱动的`Connect`方法，创建连接
	ci, err := db.connector.Connect(ctx)
    // 对db.numOpen的操作需要上锁
	db.mu.Lock()
	defer db.mu.Unlock()
    // 如果数据库已经关闭
	if db.closed {
        // 将成功创建的连接关闭
		if err == nil {
			ci.Close()
		}
		db.numOpen--
		return
	}
    // 如果创建连接失败
	if err != nil {
		db.numOpen--
		db.putConnDBLocked(nil, err)
        // 向db.openerCh中send信号
		db.maybeOpenNewConnections()
		return
	}
    // 创建成功，封装一个driverConn
	dc := &driverConn{
		db:         db,
		createdAt:  nowFunc(),
		returnedAt: nowFunc(),
		ci:         ci,
	}
	if db.putConnDBLocked(dc, err) {
		// 如果加入连接池成功，则要添加依赖关系
		db.addDepLocked(dc, dc)
	} else {
		db.numOpen--
		ci.Close()
	}
}

// /src/database/sql/sql.go:699
func (db *DB) addDepLocked(x finalCloser, dep any) {
	if db.dep == nil {
		db.dep = make(map[finalCloser]depSet)
	}
    
	xdep := db.dep[x]
	if xdep == nil {
		xdep = make(depSet)
		db.dep[x] = xdep
	}
	xdep[dep] = true
}

创建成功和失败都调用了db.putConnDBLocked，其作用是将连接添加到连接池

// /src/database/sql/sql.go:1498
// If err != nil, the value of dc is ignored.
// If err == nil, then dc must not equal nil.
// If a connRequest was fulfilled or the *driverConn was placed in the
// freeConn list, then true is returned, otherwise false is returned.
func (db *DB) putConnDBLocked(dc *driverConn, err error) bool {
    // 如果db已经关闭或者连接池数量到达上限了，直接返回false
	if db.closed {
		return false
	}
	if db.maxOpen > 0 && db.numOpen > db.maxOpen {
		return false
	}
    // 如果正在请求连接，取出一个请求并将当前的dc和err返回给它
	if c := len(db.connRequests); c > 0 {
		var req chan connRequest
		var reqKey uint64
        
		for reqKey, req = range db.connRequests {
			break
		}
		delete(db.connRequests, reqKey) // Remove from pending requests.
		if err == nil {
			dc.inUse = true
		}
		req <- connRequest{
			conn: dc,
			err:  err,
		}
		return true
	} else if err == nil && !db.closed {
        // 如果空闲连接池还有容量则加入
		if db.maxIdleConnsLocked() > len(db.freeConn) {
			db.freeConn = append(db.freeConn, dc)
			db.startCleanerLocked()
			return true
		}
        // 如果没容量了只能返回false，该连接将会被关闭
		db.maxIdleClosed++
	}
	return false
}

// /src/database/sql/sql.go:1699
func (db *DB) QueryContext(ctx context.Context, query string, args ...any) (*Rows, error) {
	var rows *Rows
	var err error
	
    // 重试机制，在请求过程中，可能会因为连接过期而导致发生偶发性的 ErrBadConn 错误，
    // 针对这种错误，可以采用重试的方式来提高请求的成功率.
	err = db.retry(func(strategy connReuseStrategy) error {
		rows, err = db.query(ctx, query, args, strategy)
		return err
	})

	return rows, err
}


// 最多可以因为 BadConn 类型的错误重试两次
const maxBadConnRetries = 2

func (db *DB) retry(fn func(strategy connReuseStrategy) error) error {
	for i := int64(0); i < maxBadConnRetries; i++ {
		err := fn(cachedOrNewConn)
		// retry if err is driver.ErrBadConn
		if err == nil || !errors.Is(err, driver.ErrBadConn) {
			return err
		}
	}

	return fn(alwaysNewConn)
}

重试2次后，将会指定创建新连接进行查询，查询将会调用db.query：

func (db *DB) query(ctx context.Context, query string, args []any, strategy connReuseStrategy) (*Rows, error) {
    // 获取连接
	dc, err := db.conn(ctx, strategy)
	if err != nil {
		return nil, err
	}
    
	return db.queryDC(ctx, nil, dc, dc.releaseConn, query, args)

来看看获取连接的具体实现细节：

// /src/database/sql/sql.go:1281
func (db *DB) conn(ctx context.Context, strategy connReuseStrategy) (*driverConn, error) {
	db.mu.Lock()
	if db.closed {
		db.mu.Unlock()
		return nil, errDBClosed
	}
	// Check if the context is expired.
	select {
	default:
	case <-ctx.Done():
		db.mu.Unlock()
		return nil, ctx.Err()
	}
	lifetime := db.maxLifetime

	// 优先从空闲连接池获取，优先取尾部，即最后加入的连接
	last := len(db.freeConn) - 1
	if strategy == cachedOrNewConn && last >= 0 {
		// Reuse the lowest idle time connection so we can close
		// connections which remain idle as soon as possible.
		conn := db.freeConn[last]
		db.freeConn = db.freeConn[:last]
		conn.inUse = true
        // 如果连接已经过期，计数、关闭该连接并返回错误
		if conn.expired(lifetime) {
			db.maxLifetimeClosed++
			db.mu.Unlock()
			conn.Close()
			return nil, driver.ErrBadConn
		}
		db.mu.Unlock()

        // 如果需要的话，reset session，具体行为由第三方driver自行实现
		if err := conn.resetSession(ctx); errors.Is(err, driver.ErrBadConn) {
			conn.Close()
			return nil, err
		}

		return conn, nil
	}

    // 没有空闲连接
    // db.numOpen >= db.maxOpen意味着不能再创建新的连接了，只能等待有连接释放时被唤醒
	if db.maxOpen > 0 && db.numOpen >= db.maxOpen {
		// 
		req := make(chan connRequest, 1)
		reqKey := db.nextRequestKeyLocked()
		db.connRequests[reqKey] = req
		db.waitCount++
		db.mu.Unlock()

		waitStart := nowFunc()

		// Timeout the connection request with the context.
		select {
		case <-ctx.Done():
			db.mu.Lock()
			delete(db.connRequests, reqKey)
			db.mu.Unlock()

			db.waitDuration.Add(int64(time.Since(waitStart)))
			
           // 如果在delete时已经返回连接了，需要将该连接放回连接池
			select {
			default:
			case ret, ok := <-req:
				if ok && ret.conn != nil {
					db.putConn(ret.conn, ret.err, false)
				}
			}
			return nil, ctx.Err()
		case ret, ok := <-req:
			db.waitDuration.Add(int64(time.Since(waitStart)))

			if !ok {
				return nil, errDBClosed
			}

            // 只有策略是cachedOrNewConns时检查连接是否过期
            // 如果我们是指定要新的连接，不需要关注获取到的连接过期时间，因为你无法保证，可能做完判断的下一刻连接就过期了
			if strategy == cachedOrNewConn && ret.err == nil && ret.conn.expired(lifetime) {
				db.mu.Lock()
				db.maxLifetimeClosed++
				db.mu.Unlock()
				ret.conn.Close()
				return nil, driver.ErrBadConn
			}
			if ret.conn == nil {
				return nil, ret.err
			}

			// Reset the session if required.
			if err := ret.conn.resetSession(ctx); errors.Is(err, driver.ErrBadConn) {
				ret.conn.Close()
				return nil, err
			}
			return ret.conn, ret.err
		}
	}
    
	// 连接数未达上限，则会调用第三方驱动的 connector 完成新连接的创建
	db.numOpen++ // optimistically
	db.mu.Unlock()
	ci, err := db.connector.Connect(ctx)
	if err != nil {
		db.mu.Lock()
		db.numOpen-- // correct for earlier optimism
		db.maybeOpenNewConnections()
		db.mu.Unlock()
		return nil, err
	}
	db.mu.Lock()
	dc := &driverConn{
		db:         db,
		createdAt:  nowFunc(),
		returnedAt: nowFunc(),
		ci:         ci,
		inUse:      true,
	}
	db.addDepLocked(dc, dc)
	db.mu.Unlock()
	return dc, nil
}

获取连接之后，执行查询，queryDC会依赖于第三方驱动完成请求的执行：

  1. 通过连接将sql预处理为statement
  2. 执行请求，返回结果
  3. 回收或释放连接
// /src/database/sql/sql.go:1733
func (db *DB) queryDC(ctx, txctx context.Context, dc *driverConn, releaseConn func(error), query string, args []any) (*Rows, error) {
    // driver.QueryerContext由第三方自行实现
	queryerCtx, ok := dc.ci.(driver.QueryerContext)
	var queryer driver.Queryer
	if !ok {
		queryer, ok = dc.ci.(driver.Queryer)
	}
	if ok {
		var nvdargs []driver.NamedValue
		var rowsi driver.Rows
		var err error
		withLock(dc, func() {
           // 处理参数
			nvdargs, err = driverArgsConnLocked(dc.ci, nil, args)
			if err != nil {
				return
			}
           // 执行查询，获得结果
			rowsi, err = ctxDriverQuery(ctx, queryerCtx, queryer, query, nvdargs)
		})
		if err != driver.ErrSkip {
			if err != nil {
				releaseConn(err)
				return nil, err
			}
			// Note: ownership of dc passes to the *Rows, to be freed
			// with releaseConn.
			rows := &Rows{
				dc:          dc,
				releaseConn: releaseConn,
				rowsi:       rowsi,
			}
			rows.initContextClose(ctx, txctx)
			return rows, nil
		}
	}

	var si driver.Stmt
	var err error
	withLock(dc, func() {
		si, err = ctxDriverPrepare(ctx, dc.ci, query)
	})
	if err != nil {
		releaseConn(err)
		return nil, err
	}

	ds := &driverStmt{Locker: dc, si: si}
	rowsi, err := rowsiFromStatement(ctx, dc.ci, ds, args...)
	if err != nil {
		ds.Close()
		releaseConn(err)
		return nil, err
	}

	// Note: ownership of ci passes to the *Rows, to be freed
	// with releaseConn.
	rows := &Rows{
		dc:          dc,
		releaseConn: releaseConn,
		rowsi:       rowsi,
		closeStmt:   ds,
	}
	rows.initContextClose(ctx, txctx)
	return rows, nil
}
